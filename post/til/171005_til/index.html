function foo(i) {
  if (i < 0)
    return;
  console.log('begin:' + i);
  foo(i - 1);
  console.log('end:' + i);
}
foo(3);

// Output:
// begin:3 출력 & foo(2) 실행 & foo(3) 남은명령 -> 스택1
// begin:2 출력 & foo(1) 실행 & foo(2) 남은명령 -> 스택2
// begin:1 출력 & foo(0) 실행 & foo(-1) 남은명령 -> 스택3
// begin:0 출력 & foo(-1) 실행 즉시 탈출 & 남은명령 -> 스택4
